Number Guessing Game

from random import randint # Generate a random number random_number = randint(1, 100) def check_guess(): user_guess = int(Element("guess").element.value) result_element = Element("result") if user_guess < random_number: result_element.write("Too low! Try again.") elif user_guess > random_number: result_element.write("Too high! Try again.") else: result_element.write("Correct! You guessed the number.") # Attach the function to the button Element("guessButton").element.onclick = check_guess
